1. المصادقة والمستخدمين

▼

POST تسجيل مستخدم جديد

{{baseUrl}}/auth/register

الحقلالنوعمطلوبالوصف
namestringنعمالاسم الكامل
emailemailنعمالبريد الإلكتروني (فريد)
passwordstringنعمكلمة المرور (min: 8)
password_confirmationstringنعمتأكيد كلمة المرور
phonestringنعمرقم الهاتف
rolestringلاuser أو owner
ملاحظة
يرجى التحقق من البريد الإلكتروني لتوثيق الحساب.
{
    "name": "اسم المستخدم",
    "email": "user@example.com",
    "password": "password123",
    "password_confirmation": "password123",
    "phone": "966501234567",
    "role": "user"
}

{
    "success": true,
    "message": "تم التسجيل بنجاح...",
    "data": {
        "user": {
            "id": 1,
            "name": "اسم المستخدم",
            "email": "user@example.com",
            "phone": "966501234567",
            "role": "user",
            "is_active": true,
            "created_at": "2026-03-15T12:00:00"
        }
    }
}

2. المناطق (Regions)

▼

GET جلب جميع المناطق مع التصفية والصفحات

{{baseUrl}}/regions

Parameterالوصفمثال
typeنوع المنطقة?type=city
parent_idالمنطقة الأب?parent_id=1 أو ?parent_id=null
searchالبحث بالاسم?search=دمشق
has_childrenفلترة حسب وجود أبناء?has_children=true
sort_byحقل الترتيب?sort_by=name
sort_orderاتجاه الترتيب?sort_order=asc
per_pageعدد بالصفحة?per_page=10

أنواع المناطق: country, governorate, city, neighborhood

{
    "success": true,
    "data": [
        {
            "id": 1,
            "name": "سوريا",
            "type": "country",
            "parent_id": null,
            "created_at": "2026-03-15T12:00:00"
        }
    ]
}

3. العقارات (Properties)

▼

GET جلب قائمة العقارات المنشورة

{{baseUrl}}/properties

Parameterالوصفمثال
typeنوع العقار (sale/rent)?type=sale
property_typeنوع العقار?property_type=apartment
region_idمعرف المنطقة?region_id=4
