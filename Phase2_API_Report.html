📝 ملخص التعديلات

✅ الجديد:

  • الدخول برقم الهاتف أو البريد الإلكتروني
  • إزالة التحقق من توثيق البريد الإلكتروني
  • رسالة تسجيل بسيطة بدون ذكر التوثيق

❌ المحذوف:

  • التحقق من email_verified_at عند تسجيل الدخول
  • رسالة "لم يتم توثيق حسابك بعد"

📝 1. AuthService.php

login() معدّل
app/Services/AuthService.php

قبل:

// البحث بـ email فقط $user = User::where('email', $data['email'])->first(); // ... // ❌ التحقق من توثيق البريد if (is_null($user->email_verified_at)) { return 'unverified'; }

بعد:

// البحث بـ email أو phone $loginField = $data['email']; $user = User::where('email', $loginField) ->orWhere('phone', $loginField) ->first(); // ✅ تم إزالة التحقق من email_verified_at

📝 2. AuthRequest.php

login rules معدّل
app/Http/Requests/AuthRequest.php

قبل:

'login' => [ 'email' => 'required|string|email', // فقط email 'password' => 'required|string', ],

بعد:

'login' => [ 'email' => 'required|string|min:8|max:50', // email أو phone 'password' => 'required|string', ],

📄 توثيق الـ API

POST /auth/register

الحقل النوع الحالة
name string مطلوب
phone string مطلوب
password string مطلوب
password_confirmation string مطلوب
email string اختياري

POST /auth/login

الحقل النوع الوصف
email string البريد الإلكتروني أو رقم الهاتف
password string كلمة المرور

أمثلة:

// تسجيل الدخول بالبريد الإلكتروني { "email": "user@example.com", "password": "password123" } // تسجيل الدخول برقم الهاتف { "email": "0912345678", "password": "password123" } // تسجيل الدخول مع رمز الدولة { "email": "+963912345678", "password": "password123" }
